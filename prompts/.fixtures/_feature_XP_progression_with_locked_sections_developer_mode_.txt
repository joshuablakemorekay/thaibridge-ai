XP_LEVELS = {1: 0, 2: 100, 3: 250, 4: 500, 5: 1000}

def add_xp(progress, amount, action=""):
    """Award XP (multiplied by subscription tier) and detect level-ups."""
    multiplier = {"free": 1, "basic": 2, "pro": 3}.get(progress.get("subscription", "free"), 1)
    progress["xp"] = progress.get("xp", 0) + amount * multiplier
    return progress
